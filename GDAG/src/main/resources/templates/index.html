# Dockerfile

Multi-stage 
FROM eclipse-temurin:21-jdk AS build
WORKDIR /app
COPY . .
RUN ./gradlew bootJar

FROM eclipse-temurin:21-jre-alpine
COPY --from=build /app/*.jar app.jar
ENTRYPOINT ["java", "-jar", "app.jar"]
